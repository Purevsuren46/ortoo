// API Key авах
curl -X POST https://api.ortoo.mn/api/keys \
  -H "Content-Type: application/json" \
  -d '{"tier":"free"}'

// Ханш авах
curl https://api.ortoo.mn/api/v1/rates \
  -H "X-API-Key: ort_your_key"

// Хөрвүүлэх
curl -X POST https://api.ortoo.mn/api/v1/convert \
  -H "X-API-Key: ort_your_key" \
  -d '{"amount":1000,"from":"usd"}'
// Node.js
const res = await fetch('https://api.ortoo.mn/api/v1/rates/usd', {
  headers: { 'X-API-Key': 'ort_your_key' }
});
const data = await res.json();
// data.banks → [{ name, mn, buy, sell, spread }]
